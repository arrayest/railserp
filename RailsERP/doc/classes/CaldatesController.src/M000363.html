  def new
    @caldate = Caldate.new
    @contacts= Contact.find(:all)
    @datearts=Dateart.find(:all)
    @dateprios=Dateprio.find(:all)
    @datestatuses=Datestatus.find(:all)

    respond_to do |format|
      format.html # new.html.erb
      format.xml  { render :xml => @caldate }
    end
  end
