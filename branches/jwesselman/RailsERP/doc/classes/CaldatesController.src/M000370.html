   def datechange
   @today = params[:today]
   @today=@today.to_date
   @caldates = Caldate.find(:all, :limit => 20)

   respond_to do |format|
      format.html { render :template =>"caldates/index"}
      format.xml  { render :xml => @caldates }
    end
  end
