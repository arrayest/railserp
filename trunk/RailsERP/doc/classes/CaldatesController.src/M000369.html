 def filterdate
   @today = params[:today]
   @today=@today.to_date
   @date=params[:date]
   @date=@date.to_date
   @caldates = Caldate.find(:all, :conditions =>["startdate> ? and startdate< ?",@date,@date+1])
   respond_to do |format|
      format.html { render :template =>"caldates/index"}
      format.xml  { render :xml => @caldates }
    end
 end
